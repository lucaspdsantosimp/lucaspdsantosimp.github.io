const PersonalizeDay = ({ originalDetails }) => {

    const [prefs, setPrefs] = useState('');

    const [personalizedPlan, setPersonalizedPlan] = useState('');

    const [isLoading, setIsLoading] = useState(false);

    const [error, setError] = useState('');


    const handlePersonalize = async () => {

        setIsLoading(true);

        setError('');

        setPersonalizedPlan('');


        const prompt = `Você é um planejador de viagens especialista na Serra Gaúcha. Abaixo está o roteiro original para um dia de viagem. O usuário tem as seguintes preferências: '${prefs}'. Por favor, reescreva o roteiro para este dia, incorporando as preferências do usuário. Mantenha o formato de 'Manhã', 'Tarde' e 'Noite', mas sugira atividades e locais alternativos ou adicionais que se alinhem com os interesses do usuário. O roteiro original é: Manhã: ${originalDetails.morning.description}, Tarde: ${originalDetails.afternoon.description}, Noite: ${originalDetails.night.description}. Apresente a resposta de forma clara e organizada, usando markdown com **negrito** para os títulos (Manhã, Tarde, Noite) e listas com hífens para as atividades.`;

        

        try {

            let chatHistory = [{ role: "user", parts: [{ text: prompt }] }];

            const payload = { contents: chatHistory };

            const apiKey = ""; 

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

            const response = await fetch(apiUrl, {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify(payload)

            });


            if (!response.ok) {

                throw new Error(`API error: ${response.statusText}`);

            }


            const result = await response.json();

            

            if (result.candidates && result.candidates.length > 0 && result.candidates[0].content && result.candidates[0].content.parts && result.candidates[0].content.parts.length > 0) {

                const text = result.candidates[0].content.parts[0].text;

                setPersonalizedPlan(text);

            } else {

                throw new Error("Não foi possível gerar um plano personalizado. Tente novamente.");

            }

        } catch (e) {

            setError(e.message);

        } finally {

            setIsLoading(false);

        }

    };


    return (

        <div className="mt-12 p-6 bg-gradient-to-br from-emerald-50 to-teal-50 rounded-2xl shadow-lg border border-emerald-200">

            <h3 className="text-2xl font-bold text-emerald-900 font-serif mb-4 flex items-center">

                <Sparkles className="mr-3 text-yellow-500" />

                Personalize seu Dia

            </h3>

            <p className="text-gray-600 mb-4">Diga-nos do que você gosta (ex: "aventura e natureza", "comida e vinho", "cultura e história") e a IA criará um plano especial para você.</p>

            <div className="flex flex-col sm:flex-row gap-2">

                <input

                    type="text"

                    value={prefs}

                    onChange={(e) => setPrefs(e.target.value)}

                    placeholder="Quais são suas preferências?"

                    className="flex-grow p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-emerald-500 focus:outline-none"

                    disabled={isLoading}

                />

                <button

                    onClick={handlePersonalize}

                    disabled={isLoading || !prefs}

                    className="bg-emerald-600 text-white font-bold py-3 px-6 rounded-lg hover:bg-emerald-700 transition-colors disabled:bg-gray-400 disabled:cursor-not-allowed flex items-center justify-center"

                >

                    {isLoading ? <Loader2 className="animate-spin mr-2" /> : <BrainCircuit className="mr-2" />}

                    Gerar Roteiro ✨

                </button>

            </div>


            {isLoading && (

                <div className="mt-6 text-center">

                    <Loader2 className="h-8 w-8 text-emerald-600 animate-spin mx-auto" />

                    <p className="mt-2 text-gray-600">Criando um dia perfeito para você...</p>

                </div>

            )}

            {error && <div className="mt-6 p-4 bg-red-100 text-red-700 rounded-lg">{error}</div>}

            {personalizedPlan && (

                <div className="mt-6 p-6 bg-white rounded-lg prose max-w-none" dangerouslySetInnerHTML={{ __html: personalizedPlan.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>').replace(/\n/g, '<br />').replace(/- (.*?)(<br \/>|$)/g, '<p style="margin-left: 20px;">&bull; $1</p>') }}>

                </div>

            )}

        </div>

    );

};

const RestaurantCard = ({ name, specialty }) => {

    const [isModalOpen, setIsModalOpen] = useState(false);

    const [suggestion, setSuggestion] = useState('');

    const [isLoading, setIsLoading] = useState(false);

    const [error, setError] = useState('');


    const getSuggestion = async () => {

        if (suggestion) { // Don't fetch if already have a suggestion

            setIsModalOpen(true);

            return;

        }

        setIsModalOpen(true);

        setIsLoading(true);

        setError('');


        const prompt = `Você é um crítico gastronômico. Para o restaurante '${name}', que é conhecido por '${specialty}', sugira 2 ou 3 pratos que um casal poderia experimentar, com uma breve e atrativa descrição de cada um. Apresente como uma lista com hífens.`;

        

        try {

            let chatHistory = [{ role: "user", parts: [{ text: prompt }] }];

            const payload = { contents: chatHistory };

            const apiKey = "";

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

            const response = await fetch(apiUrl, {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify(payload)

            });


            if (!response.ok) throw new Error(`API error: ${response.statusText}`);


            const result = await response.json();

            if (result.candidates && result.candidates[0]?.content?.parts[0]?.text) {

                setSuggestion(result.candidates[0].content.parts[0].text);

            } else {

                throw new Error("Não foi possível obter sugestões. Tente novamente.");

            }

        } catch (e) {

            setError(e.message);

        } finally {

            setIsLoading(false);

        }

    };


    return (

        <>

            <div className="bg-amber-50 p-4 rounded-lg border border-amber-200 flex flex-col justify-between">

                <div>

                    <h4 className="font-bold text-amber-900">{name}</h4>

                    <p className="text-sm text-amber-700">{specialty}</p>

                </div>

                <button onClick={getSuggestion} className="mt-3 bg-amber-500 text-white text-xs font-bold py-1 px-3 rounded-full hover:bg-amber-600 transition-colors self-start flex items-center">

                    <Sparkles className="mr-1 h-3 w-3"/> Sugestão do Chef

                </button>

            </div>

            {isModalOpen && (

                <div className="fixed inset-0 bg-black bg-opacity-50 flex justify-center items-center p-4 z-50">

                    <div className="bg-white rounded-lg shadow-2xl p-6 w-full max-w-md relative">

                        <button onClick={() => setIsModalOpen(false)} className="absolute top-2 right-2 text-gray-400 hover:text-gray-600">

                            <X className="h-6 w-6" />

                        </button>

                        <h3 className="text-2xl font-bold text-amber-800 mb-2">{name}</h3>

                        <p className="text-gray-500 mb-4">Especialidade: {specialty}</p>

                        <div className="border-t pt-4">

                            {isLoading && <div className="flex justify-center items-center"><Loader2 className="h-6 w-6 animate-spin text-amber-500" /> <span className="ml-2">Buscando sugestões...</span></div>}

                            {error && <div className="text-red-600">{error}</div>}

                            {suggestion && <div className="prose prose-sm max-w-none" dangerouslySetInnerHTML={{ __html: suggestion.replace(/\n/g, '<br />') }}></div>}

                        </div>

                    </div>

                </div>

            )}

        </>

    );

};


const Footer = () => (

    <footer className="bg-gray-800 text-white p-4 text-center text-sm">

        <p>Roteiro criado com carinho para uma viagem inesquecível. Boa viagem!</p>

    </footer>

);

  );

}
